    COUNT(*) AS total,
    SUM(CASE WHEN Loan_Status = 'Y' THEN 1 ELSE 0 END) AS approved,

    ROUND(100* SUM(CASE WHEN Loan_Status = 'Y' THEN 1 ELSE 0 END) / COUNT(*), 2) AS approval_pct
FROM Loan_approval_dataset


--Gender wise approval rates 
SELECT GENDER,COUNT(*),
	SUM(CASE WHEN Loan_Status = 'Y' THEN 1 ELSE 0 END) AS Approved ,
    SUM(CASE WHEN Loan_Status = 'N' THEN 1 ELSE 0 END) AS Notapproved,
	ROUND(100* SUM(CASE WHEN Loan_Status = 'Y' THEN 1 ELSE 0 END) / COUNT(*), 2) AS approval_pct
FROM Loan_approval_dataset
GROUP BY GENDER

--Both genders have similar approval rates — Female 66.9% vs Male 69.1%.
--Gender doesn't seem to be a major factor!


--EDUCATION WISE APPROVAL

--Approval Rate by Risk Tier
WITH RiskCTE AS (
    SELECT 
        Loan_Status,
        CASE
            WHEN Credit_History = 1 AND LoanAmount < 150 AND ApplicantIncome > 5000 THEN 'Low Risk'
            WHEN Credit_History = 1 AND LoanAmount BETWEEN 150 AND 300 THEN 'Medium Risk'
            WHEN Credit_History = 0 OR LoanAmount > 300 THEN 'High Risk'
            ELSE 'Uncategorized'
        END AS risk_tier
    FROM Loan_approval_dataset
)
SELECT 
    risk_tier,
    COUNT(*) AS total,
    SUM(CASE WHEN Loan_Status = 'Y' THEN 1 ELSE 0 END) AS approved,
    ROUND(100 * SUM(CASE WHEN Loan_Status ='Y' THEN 1 ELSE 0 END) / COUNT(*), 2) AS approval_pct
FROM RiskCTE
GROUP BY risk_tier
ORDER BY approval_pct DESC;

/*High Risk applicants have only 20% approval rate vs 87% for Low Risk — a massive 67% gap
Uncategorized has 293 rows — that's a large chunk worth investigating later.*/

WITH IncomeCTE AS (
	SELECT 
		Loan_status,
		CASE 
			WHEN ApplicantIncome < 2500 THEN 'Lowerincome'
			WHEN ApplicantIncome between 2500 and 5000 THEN 'Midincome'
			WHEN ApplicantIncome between 5001 and 10000 THEN 'Uppermidincome'
			WHEN ApplicantIncome > 10000 THEN 'HighIncome'
			ELSE 'Uncategorized'
        END AS income_tier
	FROM Loan_Approval_Dataset
)
SELECT 
    income_tier,
    COUNT(*) AS total,
    SUM(CASE WHEN Loan_Status = 'Y' THEN 1 ELSE 0 END) AS approved,
    ROUND(100 * SUM(CASE WHEN Loan_Status ='Y' THEN 1 ELSE 0 END) / COUNT(*), 2) AS approval_pct
FROM IncomeCTE
GROUP BY income_tier
ORDER BY approval_pct DESC;

/*Approval rates are almost identical across all income brackets — only a 3% difference between highest and lowest 
This further confirms that income level barely influences approval — credit history is still the dominant factor*/

--High Income Applicants with Bad Credit — Do they still get rejected?
SELECT 
COUNT(*) AS total,
SUM(CASE WHEN Loan_Status = 'Y' THEN 1 ELSE 0 END) AS approved,
ROUND(100 * SUM(CASE WHEN Loan_Status ='Y' THEN 1 ELSE 0 END) / COUNT(*), 2) AS approval_pct
FROM Loan_Approval_Dataset

ApplicantIncome > (Select AVG(ApplicantIncome) FROM Loan_Approval_Dataset) and

This is the strongest proof in your entire analysis that:
"Money can't buy loan approval — your credit history can."*/
